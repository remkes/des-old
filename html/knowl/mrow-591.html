\begin{align*} J_{\frac{1}{2}}(t) \amp = \sum_{n=0}^\infty \frac{(-1)^n}{n! \Gamma(\frac{3}{2} + n))} \left( \frac{t}{2} \right)^{2n + \frac{1}{2}}\\ \Gamma \left( \frac{3}{2} + n \right) \amp = \frac{(2n+1)!}{2^{2n+1} n!} \sqrt{\pi}\\ J_{\frac{1}{2}}(t) \amp = \sum_{n=0}^\infty \frac{(-1)^n} {\frac{(2n+1)!}{2^{2n+1}n!} \sqrt{\pi}} \left( \frac{t}{2} \right)^{2n + \frac{1}{2}}\\ \amp = \sqrt{\frac{2}{\pi t}} \sum_{n=0}^\infty \frac{(-1)^n}{(2n+1)!} t^{2n+1}\\ \amp = \sqrt{\frac{2}{\pi t}} \sin t\\ J_{-\frac{1}{2}} \amp = \sqrt{\frac{2}{\pi t}} \cos t \end{align*}
